CREATE TABLE TYPES 
(
ID NUMBER NOT NULL, 
NAME VARCHAR2(255) NOT NULL, 
CAPACITY NUMBER NOT NULL, 
COST_PER_KM NUMBER NOT NULL, 
DELETED NUMBER(1) DEFAULT 0, 
CONSTRAINT TYPES_PK PRIMARY KEY (ID) ENABLE
);

CREATE TABLE CARS 
(
ID NUMBER NOT NULL, 
MODEL VARCHAR2(255) NOT NULL,
GOV_NUMBER VARCHAR2(255) NOT NULL, 
COLOR VARCHAR2(255) NOT NULL, 
TYPE_ID NUMBER NOT NULL, 
DELETED NUMBER(1) DEFAULT 0, 
CONSTRAINT CARS_PK PRIMARY KEY (ID) ENABLE
);

CREATE TABLE ORDERS 
(
ID NUMBER NOT NULL, 
TIME_ORD DATE NOT NULL, 
TIME_DONE DATE, 
START_POINT NUMBER NOT NULL, 
PASSENGERS NUMBER NOT NULL, 
DISTANCE NUMBER, 
COST NUMBER, 
PHONE NUMBER NOT NULL, 
STATUS VARCHAR2(255) NOT NULL, 
CAR_ID NUMBER, 
DELETED NUMBER(1) DEFAULT 0, 
CONSTRAINT ORDERS_PK PRIMARY KEY (ID) ENABLE
);

CREATE TABLE POINTS 
(
ID NUMBER NOT NULL, 
POINT_REF NUMBER, 
ADDR VARCHAR2(255) NOT NULL, 
DELETED NUMBER(1) DEFAULT 0, 
CONSTRAINT POINTS_PK PRIMARY KEY (ID) ENABLE
);

ALTER TABLE CARS ADD CONSTRAINT CARS_FK FOREIGN KEY (TYPE_ID) REFERENCES TYPES (ID) ENABLE;
ALTER TABLE ORDERS ADD CONSTRAINT ORDERS_FK FOREIGN KEY (CAR_ID) REFERENCES CARS (ID) ENABLE;
ALTER TABLE ORDERS ADD CONSTRAINT ORDERS_FK2 FOREIGN KEY (START_POINT) REFERENCES POINTS (ID) ENABLE;
